ПРЕАМБУЛА
Нижеследующий документ представляет собой абсолютно полную техническую спецификацию проекта «Найдук», составленную на основе всех предыдущих обсуждений, с дополнениями из актуальных источников (GitHub, технические блоги, документация PostgreSQL и Supervisor за март-апрель 2026). Документ не содержит сокращений, обобщений или «воды». Все таблицы, все воркеры, все модели, все API, все ссылки на инструменты и все контрольные связи описаны с максимальной детализацией.

1.1. Основные таблицы платформы
sql
-- =========================================================================
-- ПОЛЬЗОВАТЕЛИ (ядро)
-- =========================================================================
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    phone VARCHAR(20),
    password_hash VARCHAR(255) NOT NULL,
    name VARCHAR(100),
    city VARCHAR(100),
    role VARCHAR(20) DEFAULT 'user',          -- user, business, admin
    is_premium BOOLEAN DEFAULT FALSE,
    rating DECIMAL(3,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы: поиск по email, городу
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_city ON users(city);
CREATE INDEX idx_users_role ON users(role);

-- =========================================================================
-- КАТЕГОРИИ (иерархические)
-- =========================================================================
CREATE TABLE categories (
    id SERIAL PRIMARY KEY,
    parent_id INT DEFAULT NULL,
    name VARCHAR(100) NOT NULL,
    slug VARCHAR(100) NOT NULL UNIQUE,
    description TEXT,
    icon VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индекс для иерархических запросов
CREATE INDEX idx_categories_parent ON categories(parent_id);
CREATE INDEX idx_categories_slug ON categories(slug);

-- =========================================================================
-- ОБЪЯВЛЕНИЯ (ТОВАРЫ)
-- =========================================================================
CREATE TABLE listings (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    category_id INT REFERENCES categories(id) ON DELETE SET NULL,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(12,2),
    city VARCHAR(100),
    status VARCHAR(20) DEFAULT 'active',       -- active, sold, archived, draft, rejected
    moderation_status VARCHAR(20) DEFAULT 'pending', -- pending, approved, rejected
    moderation_comment TEXT,
    views INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Составные индексы для фильтрации и сортировки
CREATE INDEX idx_listings_user ON listings(user_id);
CREATE INDEX idx_listings_category ON listings(category_id);
CREATE INDEX idx_listings_status ON listings(status);
CREATE INDEX idx_listings_city ON listings(city);
CREATE INDEX idx_listings_created ON listings(created_at DESC);

-- =========================================================================
-- ФОТОГРАФИИ ОБЪЯВЛЕНИЙ
-- =========================================================================
CREATE TABLE listing_photos (
    id SERIAL PRIMARY KEY,
    listing_id INT NOT NULL REFERENCES listings(id) ON DELETE CASCADE,
    photo_url VARCHAR(255) NOT NULL,
    sort_order INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_photos_listing ON listing_photos(listing_id);
1.2. Реферальная система (ядро монетизации)
sql
-- =========================================================================
-- РЕФЕРАЛЬНЫЕ ОФФЕРЫ (партнёрские предложения)
-- =========================================================================
CREATE TABLE affiliates (
    id SERIAL PRIMARY KEY,
    external_id VARCHAR(255),                    -- ID в CPA-сети
    source VARCHAR(50) NOT NULL,                 -- admitad, leads, yandex, local
    name VARCHAR(255) NOT NULL,
    description TEXT,
    commission_type VARCHAR(20) DEFAULT 'cpa',   -- cpa, cps, cpl, fixed
    commission_value DECIMAL(12,2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'RUB',
    url_template TEXT NOT NULL,                  -- с плейсхолдерами {click_id}, {aff_id}
    icon_url TEXT,
    geo_availability JSONB,                      -- {"countries":["RU"], "cities":["Москва"]}
    language VARCHAR(5),                         -- ru, uk, pl, de, en
    priority INT DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE,
    
    -- Поля для умных форм (страховки, кредиты, окна)
    form_type VARCHAR(50),                       -- insurance, credit, windows, calculator
    form_fields JSONB,                           -- {"fields":[{"name":"car_brand","type":"select",...}]}
    widget_url TEXT,
    widget_script TEXT,
    
    -- Метрики (обновляются воркером)
    clicks INT DEFAULT 0,
    conversions INT DEFAULT 0,
    revenue DECIMAL(12,2) DEFAULT 0,
    epc DECIMAL(12,4) DEFAULT 0,                 -- Earnings Per Click
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы GIN для JSONB-полей (для быстрых containment-запросов)
CREATE INDEX idx_affiliates_geo ON affiliates USING GIN(geo_availability);
-- Основной индекс для ранжирования
CREATE INDEX idx_affiliates_active_epc ON affiliates(is_active, epc DESC, priority DESC);

-- =========================================================================
-- КЛИКИ ПО РЕФЕРАЛЬНЫМ ССЫЛКАМ
-- =========================================================================
CREATE TABLE affiliate_clicks (
    id SERIAL PRIMARY KEY,
    click_id VARCHAR(64) UNIQUE NOT NULL,
    affiliate_id INT NOT NULL REFERENCES affiliates(id) ON DELETE CASCADE,
    user_id INT REFERENCES users(id) ON DELETE SET NULL,
    ip VARCHAR(45),
    user_agent TEXT,
    referer TEXT,
    city VARCHAR(100),
    session_id VARCHAR(128),
    
    -- Статус конверсии
    conversion_status VARCHAR(20) DEFAULT 'pending', -- pending, confirmed, rejected
    conversion_amount DECIMAL(12,2) DEFAULT 0,
    conversion_id VARCHAR(255),
    converted_at TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы
CREATE INDEX idx_clicks_click_id ON affiliate_clicks(click_id);
CREATE INDEX idx_clicks_affiliate ON affiliate_clicks(affiliate_id);
CREATE INDEX idx_clicks_conversion ON affiliate_clicks(conversion_status);
CREATE INDEX idx_clicks_created ON affiliate_clicks(created_at DESC);

-- =========================================================================
-- ЗАРАБОТОК ВЛАДЕЛЬЦА
-- =========================================================================
CREATE TABLE owner_earnings (
    id SERIAL PRIMARY KEY,
    source VARCHAR(50) NOT NULL,                 -- admitad, leads, yandex, local
    affiliate_id INT REFERENCES affiliates(id),
    click_id VARCHAR(64) REFERENCES affiliate_clicks(click_id),
    amount DECIMAL(12,2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'RUB',
    status VARCHAR(20) DEFAULT 'pending',       -- pending, paid
    paid_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_earnings_status ON owner_earnings(status);
CREATE INDEX idx_earnings_affiliate ON owner_earnings(affiliate_id);

-- =========================================================================
-- ПРИВЯЗКА РЕФЕРАЛЬНЫХ КУКИ (для атрибуции)

CREATE TABLE referral_cookies (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id),
    session_id VARCHAR(128) NOT NULL,
    affiliate_id INT REFERENCES affiliates(id),
    click_id VARCHAR(64),
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_cookies_session ON referral_cookies(session_id);
CREATE INDEX idx_cookies_expires ON referral_cookies(expires_at);

2.2. Таблица воркеров: кто, зачем, когда, лимиты
Воркер	Файл	Очередь (Redis)	Запуск	Лимиты	Что делает
CPA Collector	cpa_collector.php	queue:cpa:*	По расписанию (cron: каждые 6ч)	1 процесс, таймаут 300с	Собирает офферы из Admitad, Leads.su, ActionPay через API. Использует OARA-подобную структуру 
Link Checker	link_checker.php	queue:link_check	Раз в сутки (cron)	1 процесс, таймаут 60с/ссылку	Проверяет HTTP-статус всех активных ссылок, деактивирует битые (404/500)
Moderation Worker	moderation_worker.php	queue:moderation	Непрерывно	1 процесс, 10с ожидание	Вызывает AI-модель (Qwen3Guard) для проверки текста, обновляет moderation_status
QA Generator	qa_generator.php	queue:qa_generate	Непрерывно	1 процесс, 60с ожидание	Генерирует ответы на неотвеченные вопросы через 24 часа (локальная LLM)
Cleanup Worker	cleanup_worker.php	queue:cleanup	Раз в сутки (cron)	1 процесс, таймаут 600с	Удаляет объявления старше 90 дней, черновики старше 7 дней, файлы-сироты
Security Worker	security_worker.php	queue:security	Раз в час (cron)	1 процесс, таймаут 300с	Анализирует IP, блокирует подозрительных (>20 действий/час), отслеживает накрутки
EPC Updater	epc_updater.php	—	Раз в сутки (cron)	1 процесс, таймаут 600с	Пересчитывает epc = revenue / clicks для каждого оффера
Auction Worker	auction_worker.php	queue:auction	Раз в минуту (cron)	1 процесс, таймаут 60с	Завершает истекшие аукционы, определяет победителя, отправляет уведомления
SEO Worker	seo_worker.php	queue:seo	Раз в сутки (cron)	1 процесс, таймаут 600с	Проверяет sitemap.xml, robots.txt, битые внутренние ссылки
Translation Worker	translation_worker.php	queue:translate	Непрерывно	1 процесс, таймаут 30с	Переводит новые объявления на другие языки через локальную LLM
2.3. Контрольные связи: кто кого контролирует
text
[CRON] ─────────────────────────────────────────────────────────────────
   │
   ├─► запускает скрипт-диспетчер (dispatch.php) каждые 5 минут
   │   └─► проверяет, не запущены ли уже задачи, и ставит в Redis-очереди
   │
   └─► Supervisor управляет ВСЕМИ воркерами:
        ├─► CPA Collector ──► вызывает API (Admitad SDK) ──► пишет в affiliates
        ├─► Link Checker ──► читает affiliates.url_template ──► обновляет is_active
        ├─► Moderation Worker ──► читает queue:moderation ──► вызывает Ollama (Qwen3Guard) ──► пишет moderation_status
        ├─► QA Generator ──► читает queue:qa_generate ──► вызывает Ollama (Mistral) ──► пишет product_answers
        ├─► Cleanup Worker ──► удаляет из listings, listing_photos, audit_log
        ├─► Security Worker ──► читает audit_log ──► пишет в blocked_ips
        ├─► EPC Updater ──► читает affiliate_clicks ──► обновляет affiliates.epc
        ├─► Auction Worker ──► читает auctions ──► обновляет status, пишет уведомления
        ├─► SEO Worker ──► генерирует sitemap.xml, проверяет ссылки
        └─► Translation Worker ──► читает queue:translate ──► вызывает Ollama ──► обновляет мультиязычные поля

РАЗДЕЛ 3. АВТОМАТИЧЕСКИЙ СБОР ОФФЕРОВ (CPA-СЕТИ)
3.1. Используемый инструмент: OARA (Open Affiliate Report Aggregator)
Ссылка: https://github.com/fubralimited/php-oara 

OARA — это PHP-библиотека, которая поддерживает более 100 CPA-сетей, включая Admitad, Leads.su, ActionPay, Amazon, eBay, ShareASale, CommissionJunction и др. 

Установка:

bash
cd /var/www/nayduk
composer require ss-repos/php-oara
3.2. Реализация воркера CPA Collector
php
<?php
// workers/cpa_collector.php
require_once __DIR__ . '/../vendor/autoload.php';
require_once __DIR__ . '/../includes/functions.php';
require_once __DIR__ . '/../services/Database.php';

use Oara\Network\Publisher\Admitad;
use Oara\Network\Publisher\LeadsSu;
use Oara\Network\Publisher\ActionPay;

$db = Database::getInstance();
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);

// Список сетей с их credentials
$networks = [
    'admitad' => [
        'class' => Admitad::class,
        'credentials' => [
            'username' => getenv('ADMITAD_USERNAME'),
            'password' => getenv('ADMITAD_PASSWORD'),
            'api_key' => getenv('ADMITAD_API_KEY')
        ]
    ],
    'leads' => [
        'class' => LeadsSu::class,
        'credentials' => [
            'username' => getenv('LEADS_USERNAME'),
            'password' => getenv('LEADS_PASSWORD')
        ]
    ],
    'actionpay' => [
        'class' => ActionPay::class,
        'credentials' => [
            'username' => getenv('ACTIONPAY_USERNAME'),
            'password' => getenv('ACTIONPAY_PASSWORD')
        ]
    ]
];

    $task = $redis->blpop("queue:cpa:{$source}", 5);
    if (!$task) continue;
    
    try {
        // Инициализация сети через OARA
        $network = new $config['class']();
        $network->login($config['credentials']);
        
        if (!$network->checkConnection()) {
            throw new Exception("Connection failed for {$source}");
        }
        
        // Получаем список мерчантов
        $merchants = $network->getMerchantList();
        
        // Получаем транзакции за последние 30 дней
        $endDate = new DateTime();
        $startDate = (new DateTime())->modify('-30 days');
        $transactions = $network->getTransactionList(
            array_column($merchants, 'cid'),
            $startDate,
            $endDate
        );
        
        // Обновляем таблицу affiliates
        foreach ($transactions as $tx) {
            $db->insertOrUpdate('affiliates', [
                'external_id' => $tx['merchantId'],
                'source' => $source,
                'name' => $tx['merchantName'] ?? $merchants[$tx['merchantId']]['name'],
                'commission_type' => 'cps',
                'commission_value' => $tx['commission'],
                'url_template' => $tx['url'] ?? '',
                'is_active' => 1,
                'updated_at' => date('Y-m-d H:i:s')
            ], 'external_id', 'source');
        }
        
        // Логируем успех
        $db->insert('audit_log', [
            'action' => 'cpa_collect',
            'details' => json_encode(['source' => $source, 'count' => count($transactions)]),
            'created_at' => date('Y-m-d H:i:s')
        ]);
        
    } catch (Exception $e) {
        // Логируем ошибку и отправляем в Telegram
        error_log("CPA Collector {$source}: " . $e->getMessage());
        sendTelegram("❌ Ошибка сбора офферов из {$source}: {$e->getMessage()}");
    }

 * Универсальный postback-обработчик для всех CPA-сетей
 * Принимает параметры: click_id, status, amount, signature

require_once __DIR__ . '/../../includes/functions.php';
require_once __DIR__ . '/../../services/Database.php';

$db = Database::getInstance();

$clickId = $_GET['click_id'] ?? $_POST['click_id'] ?? $_GET['sub1'] ?? $_POST['sub1'] ?? '';
$status = $_GET['status'] ?? $_POST['status'] ?? 'confirmed';
$amount = (float)($_GET['amount'] ?? $_POST['amount'] ?? $_GET['payout'] ?? $_POST['payout'] ?? 0);
$signature = $_GET['signature'] ?? $_POST['signature'] ?? '';
$conversionId = $_GET['conversion_id'] ?? $_POST['conversion_id'] ?? $_GET['order_id'] ?? $_POST['order_id'] ?? '';

// Проверка подписи (секретный ключ из .env)
$secretKey = getenv('CPA_SECRET_KEY');
if ($secretKey && $signature) {
    $expected = hash_hmac('sha256', $clickId . $status . $amount, $secretKey);
    if (!hash_equals($expected, $signature)) {
        http_response_code(403);
        die('Invalid signature');
    }
}

if (!$clickId) {
    http_response_code(400);
    die('Missing click_id');
}

// Обновляем клик
$db->update('affiliate_clicks', [
    'conversion_status' => $status === 'confirmed' ? 'confirmed' : ($status === 'rejected' ? 'rejected' : 'pending'),
    'conversion_amount' => $amount,
    'conversion_id' => $conversionId,
    'converted_at' => date('Y-m-d H:i:s')
], 'click_id = ?', [$clickId]);

// Если конверсия подтверждена — начисляем доход
if ($status === 'confirmed' && $amount > 0) {
    $click = $db->fetchOne("SELECT affiliate_id FROM affiliate_clicks WHERE click_id = ?", [$clickId]);
    if ($click) {
        $db->insert('owner_earnings', [

            'created_at' => date('Y-m-d H:i:s')
        ]);
        
        // Обновляем статистику оффера

        sendTelegram("💰 Конверсия! Оффер #{$click['affiliate_id']}, сумма: {$amount} ₽");

<?php
class OllamaService {
    private $host = 'http://localhost:11434';
    
    public function generate($model, $prompt, $options = []) {
        $ch = curl_init($this->host . '/api/generate');
        $payload = array_merge([
            'model' => $model,
            'prompt' => $prompt,
            'stream' => false
        ], $options);
        
        curl_setopt($ch, CURLOPT_POST, true);
        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($payload));
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_TIMEOUT, 120);
        
        $response = curl_exec($ch);
        $error = curl_error($ch);
        curl_close($ch);
        
        if ($error) {
            throw new Exception("Ollama error: $error");
        }
        
        $result = json_decode($response, true);
        return $result['response'] ?? '';
    }
    
    public function moderate($text) {
        $prompt = "Ты — AI-модератор. Проверь текст на спам, мат, оскорбления, мошенничество. Верни JSON: {\"risk_level\": \"safe|controversial|unsafe\", \"reason\": \"причина\"}\n\nТекст: $text";
        
        $response = $this->generate('qwen3guard:8b', $prompt);
        return json_decode($response, true) ?: ['risk_level' => 'safe', 'reason' => ''];
    }
    
    public function generateAnswer($question, $productDescription, $reviews = []) {
        $reviewsText = implode("\n", array_slice($reviews, 0, 5));
        $prompt = "Ты — AI-помощник на сайте объявлений. Ответь на вопрос пользователя, основываясь на описании товара и отзывах. Будь полезным и нейтральным.\n\nОписание товара: $productDescription\n\nОтзывы: $reviewsText\n\nВопрос: $question\n\nОтвет:";
        
        return $this->generate('mistral', $prompt);
    }
    
    public function generateEmbedding($text) {
        $ch = curl_init($this->host . '/api/embeddings');
        $payload = [
            'model' => 'nomic-embed-text',
            'prompt' => $text
        ];
        
        curl_setopt($ch, CURLOPT_POST, true);
        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($payload));
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        
        $response = curl_exec($ch);
        curl_close($ch);
        
        $result = json_decode($response, true);
        return $result['embedding'] ?? [];
    }
}

require_once __DIR__ . '/../includes/functions.php';
require_once __DIR__ . '/../services/Database.php';

$db = Database::getInstance();
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);
$ollama = new OllamaService();

while (true) {
    $task = $redis->blpop('queue:moderation', 60);
    if (!$task) continue;
    
    $data = json_decode($task[1], true);
    $type = $data['type']; // listing, question, review
    $id = $data['id'];
    $text = $data['text'];
    
    try {
        $result = $ollama->moderate($text);
        
        if ($result['risk_level'] === 'unsafe') {
            if ($type === 'listing') {
                $db->update('listings', [
                    'moderation_status' => 'rejected',
                    'moderation_comment' => $result['reason']
                ], 'id = ?', [$id]);
            } else {
                $db->update('product_questions', [
                    'status' => 'rejected',
                    'moderation_reason' => $result['reason']
                ], 'id = ?', [$id]);
            }
            sendTelegram("❌ Отклонено $type #$id: {$result['reason']}");
        } else {
            if ($type === 'listing') {
                $db->update('listings', ['moderation_status' => 'approved'], 'id = ?', [$id]);
            } else {
                $db->update('product_questions', ['status' => 'approved'], 'id = ?', [$id]);
            }
        }
    } catch (Exception $e) {
        error_log("Moderation error: " . $e->getMessage());
        sendTelegram("⚠️ Ошибка модерации $type #$id: {$e->getMessage()}");
    }
}

1. Общая архитектура AI-агентов
Все AI-агенты работают через Ollama (локальный сервер LLM). Каждый агент – отдельный PHP-воркер под управлением Supervisor. Агенты используют Redis для очередей и межпроцессного взаимодействия. Модели подбираются под задачи: лёгкая модерация – Qwen3Guard, генерация ответов – Mistral, сложный анализ – DeepSeek-v3.

2. Агент модерации (Moderation Worker)
Файл: /workers/moderation_worker.php
Очередь: queue:moderation
Модель: Qwen3Guard‑Gen‑8B (119 языков, выдаёт причину блокировки)
Задачи:

Проверка текста объявлений, вопросов, отзывов на спам, мат, оскорбления, мошенничество.

Проверка изображений (опционально, через мультимодальную версию модели, если доступна).

Обновление moderation_status в таблицах listings, product_questions.

Полный код воркера:

php
<?php
// workers/moderation_worker.php
require_once __DIR__ . '/../includes/functions.php';
require_once __DIR__ . '/../services/Database.php';
require_once __DIR__ . '/../services/OllamaService.php';

$db = Database::getInstance();
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);
$ollama = new OllamaService();

// Системный промпт для модерации (кастомизируется в админке)
$systemPrompt = "Ты – AI-модератор сайта объявлений. Твоя задача – определить, нарушает ли текст правила: запрещены спам, мат, оскорбления, угрозы, реклама конкурентов, мошенничество. Верни JSON с полями: risk_level (safe|controversial|unsafe) и reason (краткое объяснение).";

while (true) {
    $task = $redis->blpop('queue:moderation', 60);
    if (!$task) continue;
    
    $data = json_decode($task[1], true);
    $type = $data['type'];     // listing, question, review
    $id = $data['id'];
    $text = $data['text'];
    
    try {
        $result = $ollama->generate('qwen3guard:8b', $systemPrompt . "\n\nТекст: " . $text);
        $parsed = json_decode($result, true);
        if (!$parsed) {
            // fallback: если не JSON, считаем safe
            $risk = 'safe';
            $reason = '';
        } else {
            $risk = $parsed['risk_level'];
            $reason = $parsed['reason'];
        }
        
        if ($risk === 'unsafe') {
            if ($type === 'listing') {
                $db->update('listings', [
                    'moderation_status' => 'rejected',
                    'moderation_comment' => $reason
                ], 'id = ?', [$id]);
            } elseif ($type === 'question') {
                $db->update('product_questions', [
                    'status' => 'rejected',
                    'moderation_reason' => $reason
                ], 'id = ?', [$id]);
            }
            sendTelegram("❌ Отклонено $type #$id: $reason");
        } elseif ($risk === 'controversial') {
            // Отправляем в очередь ручной проверки
            $redis->rpush('queue:manual_review', json_encode($data));
        } else {
            if ($type === 'listing') {
                $db->update('listings', ['moderation_status' => 'approved'], 'id = ?', [$id]);
            } elseif ($type === 'question') {
                $db->update('product_questions', ['status' => 'approved'], 'id = ?', [$id]);
            }
        }
    } catch (Exception $e) {
        error_log("Moderation error: " . $e->getMessage());
        sendTelegram("⚠️ Ошибка модерации $type #$id: {$e->getMessage()}");
        // При ошибке – не блокируем, ставим на повтор через 5 минут
        $redis->rpush('queue:moderation', json_encode(array_merge($data, ['retry' => ($data['retry'] ?? 0) + 1])));
    }
}
3. Агент генерации ответов (QA Generator)
Файл: /workers/qa_generator.php
Очередь: queue:qa_generate
Модель: Mistral 7B
Задачи: Через 24 часа после вопроса, если нет ответа, генерирует ответ на основе описания товара и отзывов.

Код:

php
<?php
// workers/qa_generator.php
require_once __DIR__ . '/../services/Database.php';
require_once __DIR__ . '/../services/OllamaService.php';

$db = Database::getInstance();
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);
$ollama = new OllamaService();

while (true) {
    $task = $redis->blpop('queue:qa_generate', 60);
    if (!$task) continue;
    
    $data = json_decode($task[1], true);
    $questionId = $data['question_id'];
    
    // Получаем вопрос и товар
    $question = $db->fetchOne("SELECT * FROM product_questions WHERE id = ?", [$questionId]);
    if (!$question || $question['status'] !== 'approved') continue;
    
    // Проверяем, есть ли уже ответ
    $hasAnswer = $db->fetchOne("SELECT id FROM product_answers WHERE question_id = ?", [$questionId]);
    if ($hasAnswer) continue;
    
    $listing = $db->fetchOne("SELECT * FROM listings WHERE id = ?", [$question['listing_id']]);
    if (!$listing) continue;
    
    // Собираем отзывы (до 5)
    $reviews = $db->fetchAll("SELECT review FROM product_reviews WHERE listing_id = ? LIMIT 5", [$listing['id']]);
    $reviewsText = implode("\n", array_column($reviews, 'review'));
    
    $prompt = "Ты – AI-помощник на сайте объявлений. Ответь на вопрос пользователя, основываясь на описании товара и отзывах. Будь полезным, нейтральным, не рекламируй постороннее. Если не знаешь, скажи, что лучше уточнить у продавца.\n\nОписание: {$listing['description']}\n\nОтзывы: $reviewsText\n\nВопрос: {$question['question']}\n\nОтвет:";
    
    $answer = $ollama->generate('mistral', $prompt);
    
    // Сохраняем ответ
    $db->insert('product_answers', [
        'question_id' => $questionId,
        'user_id' => 0, // AI
        'answer' => $answer,
        'ai_generated' => true,
        'created_at' => date('Y-m-d H:i:s')
    ]);
    
    // Обновляем вопрос
    $db->update('product_questions', ['ai_answered' => true], 'id = ?', [$questionId]);
    
    sendTelegram("🤖 Сгенерирован ответ на вопрос #$questionId");
}

Файл: /workers/security_worker.php
Очередь: queue:security (запускается раз в час)
Задачи:

Анализ IP на предмет подозрительной активности (>20 объявлений/час, >50 запросов/мин).

Обнаружение накруток отзывов (много отзывов с новых аккаунтов).

Блокировка аккаунтов, если они нарушают правила.

Код:

php
<?php
// workers/security_worker.php
require_once __DIR__ . '/../services/Database.php';

$db = Database::getInstance();

// 1. Блокировка IP по количеству действий
$suspicious = $db->fetchAll("
    SELECT ip, COUNT(*) as actions
    FROM audit_log
    WHERE created_at > NOW() - INTERVAL '1 hour'
    GROUP BY ip
    HAVING COUNT(*) > 20
");
foreach ($suspicious as $row) {
    $db->insert('blocked_ips', [
        'ip' => $row['ip'],
        'expires_at' => date('Y-m-d H:i:s', strtotime('+1 day'))
    ], ['on_conflict' => 'DO NOTHING']);
    sendTelegram("🚫 Заблокирован IP {$row['ip']} за {$row['actions']} действий/час");
}

// 2. Обнаружение накруток отзывов (новые аккаунты, оставившие 5+ отзывов за сутки)
$fakeReviewers = $db->fetchAll("
    SELECT u.id, u.email, COUNT(r.id) as reviews
    FROM users u
    JOIN product_reviews r ON r.user_id = u.id
    WHERE u.created_at > NOW() - INTERVAL '7 days'
      AND r.created_at > NOW() - INTERVAL '1 day'
    GROUP BY u.id
    HAVING COUNT(r.id) > 5
");
foreach ($fakeReviewers as $user) {
    $db->update('users', ['is_blocked' => 1], 'id = ?', [$user['id']]);
    sendTelegram("🚫 Заблокирован пользователь {$user['email']} за накрутку отзывов");
}

<?php

require_once __DIR__ . '/../includes/functions.php';

$urls = [];

$static = ['', 'categories', 'blog', 'about', 'contacts'];
foreach ($static as $page) {
    $urls[] = ['loc' => "$siteUrl/$page", 'priority' => 0.8];

    $urls[] = ['loc' => "$siteUrl/category/{$cat['slug']}", 'priority' => 0.7];

1. Защита API (Rate Limiting)
Все публичные API должны иметь ограничение на количество запросов с одного IP. Используем Redis для счётчика.

Файл: /includes/rate_limit.php

php
<?php
function checkRateLimit($key, $limit = 100, $window = 60) {
    $redis = new Redis();
    $redis->connect('127.0.0.1', 6379);
    $current = $redis->incr($key);
    if ($current == 1) $redis->expire($key, $window);
    if ($current > $limit) {
        http_response_code(429);
        die(json_encode(['error' => 'Too many requests']));
    }
}
Используется в начале каждого API-скрипта:

2. CSRF-защита
Во всех формах админки и авторизации генерируется уникальный токен, который хранится в сессии.

Файл: /includes/csrf.php

<?php
function csrf_token() {
    if (empty($_SESSION['csrf_token'])) {
        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
    }
    return $_SESSION['csrf_token'];
}

function csrf_verify($token) {
    if (!isset($_SESSION['csrf_token']) || !hash_equals($_SESSION['csrf_token'], $token)) {
        http_response_code(403);
        die('Invalid CSRF token');
    }
}

Капчи не защищают от ботов. Они:

Создают фрустрацию у реальных пользователей

Не останавливают современных ботов (они или решают, или передают на распознавание)

Создают ложное чувство безопасности

Как пишут эксперты: «Вместо того чтобы заставлять людей решать головоломки, современная защита делает массовую автоматизацию дорогой и ненадёжной» .

3.4. Proof-of-Work для критических действий
Используем библиотеку PoW (можно встроить в JavaScript):

javascript
// PoW для регистрации
async function solveProofOfWork() {
    const challenge = await fetch('/api/pow/challenge').then(r => r.json());
    const start = Date.now();
    
    // Поиск nonce, хеш которого начинается с 4 нулей
    let nonce = 0;
    while (true) {
        const hash = await crypto.subtle.digest('SHA-256', 
            new TextEncoder().encode(challenge.data + nonce));
        const hashHex = Array.from(new Uint8Array(hash)).map(b => b.toString(16).padStart(2, '0')).join('');
        if (hashHex.startsWith('0000')) break;
        nonce++;
        if (Date.now() - start > 3000) break; // таймаут 3 сек
    }
    
    return nonce;
}
На сервере:

// api/pow/verify.php
function verifyPoW($data, $nonce, $difficulty = 4) {
    $hash = hash('sha256', $data . $nonce);
    return substr($hash, 0, $difficulty) === str_repeat('0', $difficulty);
}

Дополнение к спецификации «Найдук»: Режим «Отпуск» (Vacation Mode) для продавцов
1. Зачем это нужно
Вдохновение — скриншот MERCARI, где в настройках аккаунта есть пункт «Vacation». Это важная функция для площадок C2C: продавец уезжает, не может обрабатывать заказы. Если просто оставить объявления активными, покупатели будут писать, ждать ответа, а продавец не ответит — это портит рейтинг и создаёт негативный опыт. 2. Как работает в Mercari (по скриншоту)
В разделе «Settings» → «Vacation» продавец может включить режим отпуска. В Mercari это автоматически:

Скрывает объявления продавца из поиска

Убирает кнопку «Купить» на страницах товаров

Показывает уведомление «Продавец в отпуске»

Сохраняет все объявления, рейтинги, историю

При возвращении — объявления снова становятся активными

5. Преимущества для «Найдука»
Защита рейтинга продавца (не будет неотвеченных сообщений и отменённых заказов).

Удобство для покупателей — они видят статус продавца заранее.

Полная автоматизация — продавец один раз настраивает дату возвращения, и система сама восстановит объявления.

Это делает «Найдук» более продуманным, чем Авито (где такой функции нет, и продавцы просто уходят в офлайн, оставляя объявления висеть).

2.1. Структура базы данных (PostgreSQL)
sql
-- =========================================================================
-- ВАКАНСИИ (объявления работодателей)
-- =========================================================================
CREATE TABLE vacancies (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    category_id INT REFERENCES categories(id),           -- профобласть: из справочника
    title VARCHAR(255) NOT NULL,                         -- должность
    description TEXT NOT NULL,                           -- обязанности, требования
    
    -- Финансовые условия
    salary_min DECIMAL(12,2),                            -- нижняя граница
    salary_max DECIMAL(12,2),                            -- верхняя граница
    salary_type VARCHAR(20) DEFAULT 'gross',             -- gross (до вычета) / net (на руки)
    salary_currency VARCHAR(3) DEFAULT 'RUB',
    salary_period VARCHAR(20) DEFAULT 'month',           -- month, hour, day, shift
    
    -- График и условия
    employment_type VARCHAR(50) DEFAULT 'full',          -- full, part, project, internship
    schedule_type VARCHAR(50) DEFAULT 'full_day',        -- full_day, shift, flexible, remote, rotation
    experience_years INT DEFAULT 0,                      -- требуемый опыт (лет)
    education VARCHAR(100),                              -- образование: высшее, среднее и т.д.
    
    -- Локация
    city VARCHAR(100) NOT NULL,
    district VARCHAR(100),                               -- район города
    metro_station VARCHAR(100),                          -- ближайшее метро
    address TEXT,                                         -- точный адрес (опционально)
    latitude DECIMAL(10,8),                              -- для карты
    longitude DECIMAL(11,8),
    is_remote BOOLEAN DEFAULT FALSE,                     -- удалённая работа
    
    -- Компания
    company_name VARCHAR(255) NOT NULL,
    company_description TEXT,
    company_logo VARCHAR(255),
    company_website VARCHAR(255),
    
    -- Дополнительные поля (JSONB для гибкости)
    benefits JSONB,                                       -- ["дмс", "парковка", "обучение"]
    requirements JSONB,                                   -- ["права категории B", "знание 1С"]
    
    -- Статус и модерация
    status VARCHAR(20) DEFAULT 'active',                 -- active, closed, archived, draft
    moderation_status VARCHAR(20) DEFAULT 'pending',
    moderation_comment TEXT,
    
    -- Метрики
    views INT DEFAULT 0,
    responses_count INT DEFAULT 0,                       -- количество откликов
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================================================================
-- РЕЗЮМЕ СОИСКАТЕЛЕЙ
-- =========================================================================
CREATE TABLE resumes (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,                         -- желаемая должность
    description TEXT,                                     -- о себе
    
    -- Желаемые условия
    desired_salary_min DECIMAL(12,2),
    desired_salary_max DECIMAL(12,2),
    desired_salary_type VARCHAR(20) DEFAULT 'net',       -- net / gross
    desired_employment_types JSONB,                      -- ["full", "part"]
    desired_schedule_types JSONB,                        -- ["full_day", "remote"]
    
    -- Локация для поиска
    city VARCHAR(100) NOT NULL,
    ready_to_relocate BOOLEAN DEFAULT FALSE,
    ready_for_business_trips BOOLEAN DEFAULT FALSE,
    
    -- Персональные данные
    birth_date DATE,
    citizenship VARCHAR(100) DEFAULT 'РФ',
    
    -- Опыт работы (JSONB массив)
    work_experience JSONB,                               -- [{"company":"...","position":"...","start":"2020-01","end":"2023-12","description":"..."}]
    
    -- Образование (JSONB массив)
    education JSONB,                                      -- [{"institution":"...","degree":"...","year":2015}]
    
    -- Ключевые навыки (JSONB массив)
    skills JSONB,                                         -- ["php","postgresql","redis"]
    
    -- Дополнительно
    languages JSONB,                                      -- [{"lang":"english","level":"b2"}]
    about_me TEXT,
    
    -- Видимость резюме (по аналогии с Авито) [citation:1]
    visibility_mode VARCHAR(20) DEFAULT 'employers',      -- employers (всем работодателям), response_only (только при отклике), verified_only (только проверенным)
    
    -- Статус
    is_active BOOLEAN DEFAULT TRUE,
    status VARCHAR(20) DEFAULT 'active',                  -- active, hidden, archived
    
    -- Метрики
    views INT DEFAULT 0,
    invites_count INT DEFAULT 0,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================================================================
-- ОТКЛИКИ (связь вакансии и резюме)
-- =========================================================================
CREATE TABLE job_responses (
    id SERIAL PRIMARY KEY,
    vacancy_id INT NOT NULL REFERENCES vacancies(id) ON DELETE CASCADE,
    resume_id INT NOT NULL REFERENCES resumes(id) ON DELETE CASCADE,
    employer_id INT NOT NULL REFERENCES users(id),         -- владелец вакансии (для быстрых запросов)
    candidate_id INT NOT NULL REFERENCES users(id),        -- владелец резюме
    
    cover_letter TEXT,                                     -- сопроводительное письмо
    
    status VARCHAR(20) DEFAULT 'pending',                 -- pending, viewed, invited, rejected, hired
    employer_status VARCHAR(20),                          -- для фильтрации работодателем
    candidate_status VARCHAR(20),                         -- для фильтрации соискателем
    
    viewed_at TIMESTAMP,                                  -- когда работодатель просмотрел отклик
    invited_at TIMESTAMP,                                 -- когда пригласили на собеседование
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =========================================================================
-- ПРИГЛАШЕНИЯ (работодатель → соискатель без отклика)
-- =========================================================================
CREATE TABLE job_invites (
    id SERIAL PRIMARY KEY,
    vacancy_id INT NOT NULL REFERENCES vacancies(id) ON DELETE CASCADE,
    resume_id INT NOT NULL REFERENCES resumes(id) ON DELETE CASCADE,
    employer_id INT NOT NULL REFERENCES users(id),
    candidate_id INT NOT NULL REFERENCES users(id),
    message TEXT,
    status VARCHAR(20) DEFAULT 'sent',                    -- sent, viewed, accepted, declined
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    responded_at TIMESTAMP
);

-- =========================================================================
-- ИЗБРАННОЕ (сохранённые вакансии/резюме)
-- =========================================================================
CREATE TABLE job_favorites (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    vacancy_id INT REFERENCES vacancies(id) ON DELETE CASCADE,
    resume_id INT REFERENCES resumes(id) ON DELETE CASCADE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CHECK ((vacancy_id IS NOT NULL) OR (resume_id IS NOT NULL))
);

-- =========================================================================
-- ПРОВЕРЕННЫЕ РАБОТОДАТЕЛИ (бейдж)
-- =========================================================================
CREATE TABLE verified_employers (
    user_id INT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    verified_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    verification_documents JSONB,                          -- ссылки на сканы документов
    verified_by INT REFERENCES users(id)                   -- кто подтвердил (админ)
);

-- =========================================================================
-- ИНДЕКСЫ ДЛЯ БЫСТРОГО ПОИСКА
-- =========================================================================
CREATE INDEX idx_vacancies_status ON vacancies(status);
CREATE INDEX idx_vacancies_city ON vacancies(city);
CREATE INDEX idx_vacancies_category ON vacancies(category_id);
CREATE INDEX idx_vacancies_salary ON vacancies(salary_min, salary_max);
CREATE INDEX idx_vacancies_created ON vacancies(created_at DESC);

CREATE INDEX idx_resumes_user ON resumes(user_id);
CREATE INDEX idx_resumes_city ON resumes(city);
CREATE INDEX idx_resumes_skills ON resumes USING GIN(skills);
CREATE INDEX idx_resumes_active ON resumes(is_active);

CREATE INDEX idx_responses_vacancy ON job_responses(vacancy_id);

CREATE INDEX idx_responses_status ON job_responses(status);

5.3. Сортировка откликов по релевантности (аналог HeadHunter AI) 
php
// workers/job_ranking_worker.php
function rankResponses($vacancyId, $responseIds) {
    $vacancy = getVacancy($vacancyId);
    $ollama = new OllamaService();
    
    $ranked = [];
    foreach ($responseIds as $resumeId) {
        $resume = getResume($resumeId);
        $prompt = "Оцени релевантность резюме для вакансии от 1 до 10.\n\nВакансия: {$vacancy['title']}\nТребования: {$vacancy['requirements']}\n\nРезюме: {$resume['work_experience']}\nНавыки: {$resume['skills']}\n\nОценка (только число):";
        
        $score = (int)$ollama->generate('mistral', $prompt);
        $ranked[] = ['resume_id' => $resumeId, 'score' => $score];
    }
    
    usort($ranked, fn($a, $b) => $b['score'] <=> $a['score']);
    return $ranked;
}

    $vehicle = getVehicleById($vehicleId);

        'model' => $params['model'] ?? null,
        'price_min' => $params['price_min'] ?? 0,
        'price_max' => $params['price_max'] ?? 0,
        'year_min' => $params['year_min'] ?? null,
        'year_max' => $params['year_max'] ?? null,
        'mileage_max' => $params['mileage_max'] ?? null,
        'transmission' => $params['transmission'] ?? null,
        'body_type' => $params['body_type'] ?? null,
        'status' => 'pending'
    ]);
    
    // Ставим в очередь на обработку
    $redis->rpush('queue:vehicle_selection', json_encode([
        'request_id' => $requestId,
        'params' => $params
    ]));
    
    return ['request_id' => $requestId, 'message' => 'Заявка принята. Ожидайте подбора в течение 24 часов.'];
}

// workers/vehicle_selection_worker.php
function processSelectionRequest($requestId, $params) {
    $db = Database::getInstance();
    $ollama = new OllamaService();
    
    // Формируем SQL-запрос на основе параметров
    $sql = buildSearchQuery($params);
    $candidates = $db->fetchAll($sql, [], 50); // до 50 кандидатов
    
    // AI-ранжирование кандидатов
    $ranked = [];
    foreach ($candidates as $candidate) {
        $score = $ollama->generate('mistral', "Оцени релевантность автомобиля {$candidate['brand']} {$candidate['model']} для запроса: " . json_encode($params) . " от 1 до 10. Ответ только числом.");
        $ranked[] = ['id' => $candidate['id'], 'score' => (int)$score];
    }
    
    usort($ranked, fn($a, $b) => $b['score'] <=> $a['score']);
    $topIds = array_slice(array_column($ranked, 'id'), 0, 10);
    
    // Сохраняем результаты
    $db->update('vehicle_selection_requests', [
        'status' => 'completed',
        'selected_vehicles' => json_encode($topIds),
        'completed_at' => date('Y-m-d H:i:s')
    ], 'id = ?', [$requestId]);
    
    // Отправляем уведомление пользователю
    sendTelegram("🚗 Подбор авто завершён! Найдено " . count($topIds) . " вариантов.");
}
